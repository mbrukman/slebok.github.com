Syntax

csvFile ::=
	csvRecord* EOF
csvRecord ::=
	csvStringList ("\n" | EOF)
csvStringList ::=
	rawString ("," csvStringList)?
rawString ::=
	optionalSpaces (rawField optionalSpaces)?
optionalSpaces ::=
	whitespace*
whitespace ::=
	" "
	"\t"
rawField ::=
	simpleField
	quotedField
simpleField ::=
	¬("\n" | EOF | "\t" | " " | "," | """)+
quotedField ::=
	""" escapedField """
escapedField ::=
	subField (""" """ escapedField)?
subField ::=
	¬(""" | EOF)+
